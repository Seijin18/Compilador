Problemas Identificados
1. Gerenciamento de Pilha Inconsistente:

Offsets fixos (ex: SW RA, 4(FP)) sem cálculo dinâmico

Falta de ajuste do SP para variáveis locais

Sobrescrita de RA em chamadas recursivas

2. Operações Redundantes/Incorretas:

MOVE R4, R4 (no-op)

DIV R4, R4 (divisão por si mesmo)

Cálculo incorreto de a % b 
Divisão por si mesmo: DIV R4, R4 sempre resulta em quociente=1 e resto=0 (não calcula a % b).

3. Passagem de Parâmetros:

Uso incorreto de registradores para parâmetros

Falta de carregamento de valores reais

4. Gerenciamento de Registradores:

Registradores críticos (R1) sendo sobrescritos

Falta de preservação de valores entre chamadas

5. Controle de Fluxo:

Instruções BEQ/BNE com operandos iguais

Labels não utilizados (L0, L1)

// Função revisada para prólogo
void generate_function_prologue(int local_size) {
    // Salvar RA e FP
    add_instruction("SW", OP_SW, SP, RA, 0, -1, NULL);
    add_instruction("SW", OP_SW, SP, FP, 0, -2, NULL);
    
    // Ajustar SP e estabelecer novo FP
    add_instruction("SUBI", OP_SUBI, SP, SP, 0, 2, NULL);
    add_instruction("MOVE", OP_MOVE, SP, 0, FP, 0, NULL);
    
    // Alocar espaço para variáveis locais
    if(local_size > 0) {
        add_instruction("SUBI", OP_SUBI, SP, SP, 0, local_size, NULL);
    }
}

// Função revisada para epílogo
void generate_function_epilogue(int local_size) {
    // Restaurar SP
    add_instruction("MOVE", OP_MOVE, FP, 0, SP, 0, NULL);
    
    // Restaurar registradores salvos
    add_instruction("ADDI", OP_ADDI, SP, SP, 0, local_size + 2, NULL);
    add_instruction("LW", OP_LW, SP, FP, 0, -2, NULL);
    add_instruction("LW", OP_LW, SP, RA, 0, -1, NULL);
}

void process_array_access(const char* array_name, const char* index_var, const char* result_var, int is_store) {
    int base_reg = load_variable_to_register(array_name, current_function);
    int index_reg = load_variable_to_register(index_var, current_function);
    int temp_reg = get_temp_register();
    
    // Calcular deslocamento: index * sizeof(int)
    add_instruction("SLL", OP_SLL, index_reg, temp_reg, 0, 2, NULL);
    add_instruction("ADD", OP_ADD, base_reg, temp_reg, temp_reg, 0, NULL);
    
    if (is_store) {
        int value_reg = load_variable_to_register(result_var, current_function);
        add_instruction("SW", OP_SW, temp_reg, value_reg, 0, 0, NULL);
    } else {
        int dest_reg = get_register_for_variable(result_var, current_function);
        add_instruction("LW", OP_LW, temp_reg, dest_reg, 0, 0, NULL);
    }
    
    free_register(temp_reg);
}

int get_temp_register() {

        if (!registers[i].is_busy) {
            registers[i].is_busy = 1;
            return i;
        }
